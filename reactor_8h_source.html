472 reaction_status_t status; // Indicator of whether the reaction is inactive, queued, or running. RUNTIME.

485 // execution. COMMON.
486 bool is_a_control_reaction; // Indicates whether this reaction is a control reaction. Control
487 // reactions will not set ports or actions and don't require scheduling
488 // any output reactions. Default is false.
489 size_t worker_affinity; // The worker number of the thread that scheduled this reaction. Used
490 // as a suggestion to the scheduler.
491 char* name; // If logging is set to LOG or higher, then this will

493 // the reaction number.
494 reactor_mode_t* mode; // The enclosing mode of this reaction (if exists).
495 // If enclosed in multiple, this will point to the innermost mode.

518struct trigger_t {
519 reaction_t** reactions; // Array of pointers to reactions sensitive to this trigger.
520 int number_of_reactions; // Number of reactions sensitive to this trigger.
521 bool is_timer; // True if this is a timer (a special kind of action), false otherwise.
522 interval_t offset; // Minimum delay of an action. For a timer, this is also the maximum delay.
523 interval_t period; // Minimum interarrival time of an action. For a timer, this is also the maximal interarrival time.
524 lf_token_t* token; // Pointer to a token wrapping the payload (or NULL if there is none).
525 bool is_physical; // Indicator that this denotes a physical action.
526 event_t* last; // Pointer to the last event that was scheduled for this action.
527 lf_spacing_policy_t policy; // Indicates which policy to use when an event is scheduled too early.
528 size_t element_size; // The size of the payload, if there is one, zero otherwise.
529 // If the payload is an array, then this is the size of an element of the array.
530 port_status_t status; // Determines the status of the port at the current logical time. Therefore, this
531 // value needs to be reset at the beginning of each logical time.
532 //
533 // This status is especially needed for the distributed execution because the receiver logic will need
534 // to know what it should do if it receives a message with 'intended tag = current tag' from another
535 // federate.
536 // - If status is 'unknown', it means that the federate has still no idea what the status of
537 // this port is and thus has refrained from executing any reaction that has that port as its input.
538 // This means that the receiver logic can directly inject the triggered reactions into the reaction
539 // queue at the current logical time.
540 // - If the status is absent, it means that the federate has assumed that the port is 'absent'
541 // for the current logical time. Therefore, receiving a message with 'intended tag = current tag'
542 // is an error that should be handled, for example, as a violation of the STP offset in the decentralized
543 // coordination.
544 // - Finally, if status is 'present', then this is an error since multiple
545 // downstream messages have been produced for the same port for the same logical time.
546 reactor_mode_t* mode; // The enclosing mode of this reaction (if exists).
547 // If enclosed in multiple, this will point to the innermost mode.
548#ifdef FEDERATED
549 tag_t last_known_status_tag; // Last known status of the port, either via a timed message, a port absent, or a
550 // TAG from the RTI.
551 bool is_a_control_reaction_waiting; // Indicates whether at least one control reaction is waiting for this trigger
552 // if it belongs to a network input port. Must be false by default.
553 tag_t intended_tag; // The amount of discrepency in logical time between the original intended
554 // trigger time of this trigger and the actual trigger time. This currently
555 // can only happen when logical connections are used using a decentralized coordination
556 // mechanism (@see https://github.com/icyphy/lingua-franca/wiki/Logical-Connections).
557 instant_t physical_time_of_arrival; // The physical time at which the message has been received on the network according to the local clock.
558 // Note: The physical_time_of_arrival is only passed down one level of the hierarchy. Default: NEVER.
559#endif
560};
