259struct trigger_t {
260 reaction_t** reactions; // Array of pointers to reactions sensitive to this trigger.
261 int number_of_reactions; // Number of reactions sensitive to this trigger.
262 bool is_timer; // True if this is a timer (a special kind of action), false otherwise.
263 interval_t offset; // Minimum delay of an action. For a timer, this is also the maximum delay.
264 interval_t period; // Minimum interarrival time of an action. For a timer, this is also the maximal interarrival time.
265 lf_token_t* token; // Pointer to a token wrapping the payload (or NULL if there is none).
266 bool is_physical; // Indicator that this denotes a physical action.
267 event_t* last; // Pointer to the last event that was scheduled for this action.
268 lf_spacing_policy_t policy; // Indicates which policy to use when an event is scheduled too early.
269 size_t element_size; // The size of the payload, if there is one, zero otherwise.
270 // If the payload is an array, then this is the size of an element of the array.
271 port_status_t status; // Determines the status of the port at the current logical time. Therefore, this
272 // value needs to be reset at the beginning of each logical time.
273 //
274 // This status is especially needed for the distributed execution because the receiver logic will need
275 // to know what it should do if it receives a message with 'intended tag = current tag' from another
276 // federate.
277 // - If status is 'unknown', it means that the federate has still no idea what the status of
278 // this port is and thus has refrained from executing any reaction that has that port as its input.
279 // This means that the receiver logic can directly inject the triggered reactions into the reaction
280 // queue at the current logical time.
281 // - If the status is absent, it means that the federate has assumed that the port is 'absent'
282 // for the current logical time. Therefore, receiving a message with 'intended tag = current tag'
283 // is an error that should be handled, for example, as a violation of the STP offset in the decentralized
284 // coordination.
285 // - Finally, if status is 'present', then this is an error since multiple
286 // downstream messages have been produced for the same port for the same logical time.
287 reactor_mode_t* mode; // The enclosing mode of this reaction (if exists).
288 // If enclosed in multiple, this will point to the innermost mode.
289#ifdef FEDERATED
290 tag_t last_known_status_tag; // Last known status of the port, either via a timed message, a port absent, or a
291 // TAG from the RTI.
292 bool is_a_control_reaction_waiting; // Indicates whether at least one control reaction is waiting for this trigger
293 // if it belongs to a network input port. Must be false by default.
294 tag_t intended_tag; // The amount of discrepency in logical time between the original intended
295 // trigger time of this trigger and the actual trigger time. This currently
296 // can only happen when logical connections are used using a decentralized coordination
297 // mechanism (@see https://github.com/icyphy/lingua-franca/wiki/Logical-Connections).
298 instant_t physical_time_of_arrival; // The physical time at which the message has been received on the network according to the local clock.
299 // Note: The physical_time_of_arrival is only passed down one level of the hierarchy. Default: NEVER.
300#endif
301};
