141static int lf_cond_timedwait(_lf_cond_t* cond, _lf_mutex_t* mutex, int64_t absolute_time_ns) {
142 // Convert the absolute time to a timespec.
143 // timespec is seconds and nanoseconds.
144 struct timespec timespec_absolute_time
145 = {(time_t)absolute_time_ns / 1000000000LL, (long)absolute_time_ns % 1000000000LL};
146 int return_value = 0;
147 return_value = cnd_timedwait(
148 (cnd_t*)cond,
149 (mtx_t*)mutex,
150 &timespec_absolute_time
151 );
152 switch (return_value) {
153 case thrd_timedout:
154 return_value = _LF_TIMEOUT;
155 break;
156
157 default:
158 break;
159 }
160 return return_value;
161}
