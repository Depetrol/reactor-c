102 tag_t completed; // The largest logical tag completed by the federate (or NEVER if no LTC has been received).
103 tag_t last_granted; // The maximum TAG that has been granted so far (or NEVER if none granted)
104 tag_t last_provisionally_granted; // The maximum PTAG that has been provisionally granted (or NEVER if none granted)
105 tag_t next_event; // Most recent NET received from the federate (or NEVER if none received).
106 in_transit_message_record_q_t* in_transit_message_tags; // Record of in-transit messages to this federate that are not
107 // yet processed. This record is ordered based on the time
108 // value of each message for a more efficient access.
109 fed_state_t state; // State of the federate.
110 int* upstream; // Array of upstream federate ids.
111 interval_t* upstream_delay; // Minimum delay on connections from upstream federates.
112 // Here, NEVER encodes no delay. 0LL is a microstep delay.
113 int num_upstream; // Size of the array of upstream federates and delays.
114 int* downstream; // Array of downstream federate ids.
115 int num_downstream; // Size of the array of downstream federates.
116 execution_mode_t mode; // FAST or REALTIME.
117 char server_hostname[INET_ADDRSTRLEN]; // Human-readable IP address and
118 int32_t server_port; // port number of the socket server of the federate
119 // if it has any incoming direct connections from other federates.
120 // The port number will be -1 if there is no server or if the
121 // RTI has not been informed of the port number.
122 struct in_addr server_ip_addr; // Information about the IP address of the socket
123 // server of the federate.
124 bool requested_stop; // Indicates that the federate has requested stop or has replied
125 // to a request for stop from the RTI. Used to prevent double-counting
126 // a federate when handling lf_request_stop().
127} federate_t;
