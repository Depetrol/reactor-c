103static void worker_assignments_init(size_t number_of_workers, sched_params_t* params) {
104 num_levels = params->num_reactions_per_level_size;
105 max_num_workers = number_of_workers;
106 reactions_by_worker_by_level = (reaction_t****) malloc(sizeof(reaction_t***) * num_levels);
107 num_reactions_by_worker_by_level = (size_t**) malloc(sizeof(size_t*) * num_levels);
108 num_workers_by_level = (size_t*) malloc(sizeof(size_t) * num_levels);
109 max_num_workers_by_level = (size_t*) malloc(sizeof(size_t) * num_levels);
110 for (size_t level = 0; level < num_levels; level++) {
111 size_t num_reactions = params->num_reactions_per_level[level];
112 size_t num_workers = num_reactions < max_num_workers ? num_reactions : max_num_workers;
113 max_num_workers_by_level[level] = num_workers;
114 num_workers_by_level[level] = max_num_workers_by_level[level];
115 reactions_by_worker_by_level[level] = (reaction_t***) malloc(
116 sizeof(reaction_t**) * max_num_workers

118 num_reactions_by_worker_by_level[level] = (size_t*) calloc(max_num_workers, sizeof(size_t));
119 for (size_t worker = 0; worker < max_num_workers_by_level[level]; worker++) {

147static reaction_t* get_reaction(size_t worker) {
148#ifndef FEDERATED
149 int index = lf_atomic_add_fetch(num_reactions_by_worker + worker, -1);
150 if (index >= 0) {
151 return reactions_by_worker[worker][index];
152 }
153 num_reactions_by_worker[worker] = 0;
154 return NULL;
155#else
156 // This is necessary for federated programs because reactions may be inserted into the current
157 // level.
158 int old_num_reactions;
159 int current_num_reactions = num_reactions_by_worker[worker];
160 int index;
161 do {
162 old_num_reactions = current_num_reactions;
163 if (old_num_reactions <= 0) return NULL;
164 } while (
165 (current_num_reactions = lf_val_compare_and_swap(
166 num_reactions_by_worker + worker,
167 old_num_reactions,
168 (index = old_num_reactions - 1)
169 )) != old_num_reactions
170 );
171 return reactions_by_worker[worker][index];
172#endif
173}

181static reaction_t* worker_assignments_get_or_lock(size_t worker) {
182 assert(worker >= 0);
183 // assert(worker < num_workers); // There are edge cases where this doesn't hold.
184 assert(num_reactions_by_worker[worker] >= 0);
185 reaction_t* ret;
186 while (true) {
187 if ((ret = get_reaction(worker))) return ret;
188 if (worker < num_workers) {
189 for (
190 size_t victim = (worker + 1) % num_workers;
191 victim != worker;
192 victim = (victim + 1) % num_workers
193 ) {
194 if ((ret = get_reaction(victim))) return ret;
195 }
196 }
197 worker_states_lock(worker);
198 if (!num_reactions_by_worker[worker]) {
199 return NULL;
200 }
201 worker_states_unlock(worker);
202 }
203}

209static void worker_assignments_put(reaction_t* reaction) {
210 size_t level = LF_LEVEL(reaction->index);
211 assert(reaction != NULL);
212#ifndef FEDERATED
213 assert(level > current_level || current_level == 0);
214#endif
215 assert(level < num_levels);
216 // Source: https://xorshift.di.unimi.it/splitmix64.c
217 // TODO: This is probably not the most efficient way to get the randomness that we need because
218 // it is designed to give an entire word of randomness, whereas we only need
219 // ~log2(num_workers_by_level[level]) bits of randomness.
220 uint64_t hash = (uint64_t) reaction;
221 hash = (hash ^ (hash >> 30)) * 0xbf58476d1ce4e5b9;
222 hash = (hash ^ (hash >> 27)) * 0x94d049bb133111eb;
223 hash = hash ^ (hash >> 31);
224 size_t worker = hash % num_workers_by_level[level];
225 size_t num_preceding_reactions = lf_atomic_fetch_add(
226 &num_reactions_by_worker_by_level[level][worker],
227 1
228 );
229 reactions_by_worker_by_level[level][worker][num_preceding_reactions] = reaction;
230}
