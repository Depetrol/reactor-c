223/*
224 * Atomically increment the variable that ptr points to by the given value, and return the original value of the variable.
225 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.
226 * @param value The value to be added to the variable pointed to by the ptr parameter.
227 * @return The original value of the variable that ptr points to (i.e., from before the application of this operation).
228 */
229#if defined(PLATFORM_ZEPHYR)
230#define lf_atomic_fetch_add(ptr, value) _zephyr_atomic_fetch_add((int*) ptr, value)
231#elif defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
232// Assume that an integer is 32 bits.
233#define lf_atomic_fetch_add(ptr, value) InterlockedExchangeAdd(ptr, value)
234#elif defined(__GNUC__) || defined(__clang__)
235#define lf_atomic_fetch_add(ptr, value) __sync_fetch_and_add(ptr, value)
236#else
237#error "Compiler not supported"
238#endif

257/*
258 * Atomically compare the variable that ptr points to against oldval. If the
259 * current value is oldval, then write newval into *ptr.
260 * @param ptr A pointer to a variable.
261 * @param oldval The value to compare against.
262 * @param newval The value to assign to *ptr if comparison is successful.
263 * @return True if comparison was successful. False otherwise.
264 */
265#if defined(PLATFORM_ZEPHYR)
266#define lf_bool_compare_and_swap(ptr, value, newval) _zephyr_bool_compare_and_swap((bool*) ptr, value, newval)
267#elif defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
268// Assume that a boolean is represented with a 32-bit integer.
269#define lf_bool_compare_and_swap(ptr, oldval, newval) (InterlockedCompareExchange(ptr, newval, oldval) == oldval)
270#elif defined(__GNUC__) || defined(__clang__)
271#define lf_bool_compare_and_swap(ptr, oldval, newval) __sync_bool_compare_and_swap(ptr, oldval, newval)
272#else
273#error "Compiler not supported"
