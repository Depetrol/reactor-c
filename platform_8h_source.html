196/*
197 * Atomically increment the variable that ptr points to by the given value, and return the original value of the variable.
198 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.
199 * @param value The value to be added to the variable pointed to by the ptr parameter.
200 * @return The original value of the variable that ptr points to (i.e., from before the application of this operation).
201 */
202#if defined(PLATFORM_ZEPHYR)
203#define lf_atomic_fetch_add(ptr, value) _zephyr_atomic_fetch_add((int*) ptr, value)
204#elif defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
205// Assume that an integer is 32 bits.
206#define lf_atomic_fetch_add(ptr, value) InterlockedExchangeAdd(ptr, value)
207#elif defined(__GNUC__) || defined(__clang__)
208#define lf_atomic_fetch_add(ptr, value) __sync_fetch_and_add(ptr, value)
209#else
210#error "Compiler not supported"
211#endif

230/*
231 * Atomically compare the variable that ptr points to against oldval. If the
232 * current value is oldval, then write newval into *ptr.
233 * @param ptr A pointer to a variable.
234 * @param oldval The value to compare against.
235 * @param newval The value to assign to *ptr if comparison is successful.
236 * @return True if comparison was successful. False otherwise.
237 */
238#if defined(PLATFORM_ZEPHYR)
239#define lf_bool_compare_and_swap(ptr, value, newval) _zephyr_bool_compare_and_swap((bool*) ptr, value, newval)
240#elif defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
241// Assume that a boolean is represented with a 32-bit integer.
242#define lf_bool_compare_and_swap(ptr, oldval, newval) (InterlockedCompareExchange(ptr, newval, oldval) == oldval)
243#elif defined(__GNUC__) || defined(__clang__)
244#define lf_bool_compare_and_swap(ptr, oldval, newval) __sync_bool_compare_and_swap(ptr, oldval, newval)
245#else
246#error "Compiler not supported"
247#endif
