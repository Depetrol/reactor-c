200/*
201 * Atomically increment the variable that ptr points to by the given value, and return the original value of the variable.
202 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.
203 * @param value The value to be added to the variable pointed to by the ptr parameter.
204 * @return The original value of the variable that ptr points to (i.e., from before the application of this operation).
205 */
206#if defined(PLATFORM_ZEPHYR)
207#define lf_atomic_fetch_add(ptr, value) _zephyr_atomic_fetch_add((int*) ptr, value)
208#elif defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
209// Assume that an integer is 32 bits.
210#define lf_atomic_fetch_add(ptr, value) InterlockedExchangeAdd(ptr, value)
